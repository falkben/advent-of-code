cycle = 0
register = 1


def add_cycle():
    global cycle

    if cycle % 40 == 0:
        print()
        cycle = 0

    if cycle in (register, register - 1, register + 1):
        print("#", end="")
    else:
        print(".", end="")

    cycle += 1


for line in input.splitlines():

    if line == "noop":
        add_cycle()
    else:
        val = int(line.split()[1])
        add_cycle()
        add_cycle()
        register += val

print()
