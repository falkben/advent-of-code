def scenic_score(x: int, y: int):
    height = data[(x, y)]

    # check values left
    hl = 0
    for i in range(x - 1, -1, -1):
        hl += 1
        if height <= data[i, y]:
            break

    # check values right
    hr = 0
    for i in range(x + 1, len_trees):
        hr += 1
        if height <= data[i, y]:
            break
    # check values top
    ht = 0
    for i in range(y - 1, -1, -1):
        ht += 1
        if height <= data[x, i]:
            break

    # check values bottom
    hb = 0
    for i in range(y + 1, len(input.splitlines())):
        hb += 1
        if height <= data[x, i]:
            break

    return hl * hr * ht * hb
