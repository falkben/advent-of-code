depth = 0
hor = 0

for l in input.splitlines():
    dir, amount = l.split(" ")
    amount = int(amount)
    if dir == "forward":
        hor += amount
    elif dir == "down":
        depth += amount
    else:
        depth -= amount

print(depth * hor)
